    # 类实例方法 不能 使用new关键字调用

    var Product = (function() {
  function Product(name, unitPrice, number) {
  	// if(!new.target) { // es6的 new.target十分方便，但降级却不能使用；包括 instanceof 关键字
   //    throw new TypeError('Class constructor Product cannot be invoked without "new"')
   //  }
    if(Object.getPrototypeOf(this) !== Product.prototype) { // 还记得 new 的调用细节么
      throw new TypeError('Class constructor Product cannot be invoked without "new"')
    }
    
  	this.name = name
  	this.unitPrice = unitPrice
  	this.number = number
	}

	Product.count = 0

  Object.defineProperty(this, 'totalPrice', {
    get() {
      return this.number * this.unitPrice
    },
    enumerable: false
  })

  Object.defineProperty(Product.prototype, 'totalPrice', {
    get() {
      return this.number * this.unitPrice
    },
    enumerable: false
  })

  Object.defineProperty(Product.property, 'increase', {
    value: function() {
      // new p.increase() 和 p.increase()的区别
      // 前者：this的原型指向 increase 的 prototype；
      // 后者：this的原型指向 Product.prototype
      if(Object.getPropertyOf(this) === Product.prototype.increase.prototype) {
        throw new TypeError('increase is not a constructor')
      }
      this.number++
    },
    enumerable: false
  })
	Product.prototype.increase = function() {
  	this.number++
	}

  return product
})()
